**New stored properties:**

```swift
// VisionProcessor for upload mode — @CameraActor isolated
let visionProcessor = VisionProcessor()

// AVPlayerItemVideoOutput for pixel buffer extraction
nonisolated(unsafe) private var videoOutput: AVPlayerItemVideoOutput?

// Periodic time observer token for frame polling
private var frameObserverToken: Any?
```

**New method `attachVideoOutput(to player: AVPlayer)`** — called after player is created in `handlePickerResult` (where `self.player = AVPlayer(...)` is set). Call this immediately after.

```swift
private func attachVideoOutput(to player: AVPlayer) {
    // Remove old output if any
    detachVideoOutput()

    let outputSettings: [String: Any] = [
        kCVPixelBufferPixelFormatTypeKey as String: kCVPixelFormatType_420YpCbCr8BiPlanarFullRange
    ]
    let output = AVPlayerItemVideoOutput(pixelBufferAttributes: outputSettings)
    self.videoOutput = output
    player.currentItem?.add(output)

    // Poll every ~33ms (~30fps) — matches typical video frame rate
    // Using 0.033s interval; Vision will process on @CameraActor serially
    let interval = CMTime(seconds: 0.033, preferredTimescale: 600)
    frameObserverToken = player.addPeriodicTimeObserver(forInterval: interval, queue: .main) { [weak self] time in
        MainActor.assumeIsolated {
            self?.processCurrentFrame(at: time)
        }
    }
}
```

**New method `processCurrentFrame(at time: CMTime)`** — `@MainActor`, called from periodic observer:

```swift
private func processCurrentFrame(at time: CMTime) {
    guard let output = videoOutput,
          output.hasNewPixelBuffer(forItemTime: time) else { return }

    guard let pixelBuffer = output.copyPixelBuffer(forItemTime: time, itemTimeForDisplay: nil) else { return }

    // Wrap CVPixelBuffer in CMSampleBuffer for VNSequenceRequestHandler
    var timing = CMSampleTimingInfo(duration: .invalid, presentationTimeStamp: time, decodeTimeStamp: .invalid)
    var formatDescription: CMFormatDescription?
    CMVideoFormatDescriptionCreateForImageBuffer(allocator: nil, imageBuffer: pixelBuffer, formatDescriptionOut: &formatDescription)
    guard let formatDescription else { return }

    var sampleBuffer: CMSampleBuffer?
    CMSampleBufferCreateReadyWithImageBuffer(
        allocator: nil,
        imageBuffer: pixelBuffer,
        formatDescription: formatDescription,
        sampleTiming: &timing,
        sampleBufferOut: &sampleBuffer
    )
    guard let sampleBuffer else { return }

    let processor = visionProcessor
    Task { @CameraActor in
        processor.process(sampleBuffer: sampleBuffer)
    }
}
```

**New method `detachVideoOutput()`** — cleanup when a new video loads or player resets:

```swift
private func detachVideoOutput() {
    if let token = frameObserverToken, let player {
        player.removeTimeObserver(token)
        frameObserverToken = nil
    }
    videoOutput = nil
}
```

In `handlePickerResult`, after `self.player = AVPlayer(playerItem: AVPlayerItem(url: dest))`, add:
```swift
self.attachVideoOutput(to: self.player!)
```

Also call `detachVideoOutput()` at the top of `handlePickerResult` (in the reset block) before `player = nil`.

Note on frame cadence: `hasNewPixelBuffer(forItemTime:)` returns false when paused or when the current time hasn't advanced — this naturally "holds" the last processed frame on the overlay without any extra code. The periodic observer keeps firing but Vision is skipped until playback resumes.

```swift
if let player = manager.player {
    ZStack {
        VideoPlayerView(player: player)
            .frame(maxWidth: .infinity, maxHeight: .infinity)

        // Skeleton overlay on top of video — same pattern as LiveSessionView
        if skeletonPref.isEnabled {
            GeometryReader { geo in
                SkeletonOverlayView(
                    joints: manager.visionProcessor.detectedPose?.joints ?? [:],

            .allowsHitTesting(false)
        }
    }
    .frame(maxWidth: .infinity, maxHeight: .infinity)
}
```

The `allowsHitTesting(false)` on the GeometryReader ensures the skeleton overlay does not intercept the scrubber or play/pause button taps in `VideoPlayerView`.
