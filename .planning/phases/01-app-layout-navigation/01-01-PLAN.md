## Key Architecture Contracts (established in RESEARCH.md)

AppCoordinator (the navigation source of truth):
```swift
@Observable
@MainActor
final class AppCoordinator {
    var path = NavigationPath()
    var isDrawerOpen = false

    enum Destination: Hashable {
        case history
        case upload
        case settings
        case liveSession
    }

    func navigate(to destination: Destination) {
        isDrawerOpen = false
        path.append(destination)
    }

    func popToRoot() {
        path.removeLast(path.count)
    }
}
```

Brand colors (Color extension, RESEARCH.md Pattern 4):
```swift
extension Color {
    init(hex: UInt, opacity: Double = 1.0) {
        self.init(
            red: Double((hex >> 16) & 0xFF) / 255,
            green: Double((hex >> 8) & 0xFF) / 255,
            blue: Double(hex & 0xFF) / 255,
            opacity: opacity
        )
    }
    static let brandBackground  = Color(hex: 0x0C0906)
    static let brandEmber       = Color(hex: 0xFF6B2B)
    static let brandAmber       = Color(hex: 0xFFAA3B)
    static let brandGold        = Color(hex: 0xFFD166)
    static let textPrimary      = Color(hex: 0xFAF3EC)
    static let textSecondary    = Color(hex: 0xB8A090)
}
```

Brand fonts (RESEARCH.md Pattern 5):
```swift
extension Font {
    static func mono(_ size: CGFloat) -> Font {
        .custom("JetBrainsMono-Regular", size: size)
    }
    static func monoBold(_ size: CGFloat) -> Font {
        .custom("JetBrainsMono-Bold", size: size)
    }
}
```

**JetBrains Mono font embedding:**
1. Download JetBrains Mono from https://www.jetbrains.com/lp/mono/ (or https://github.com/JetBrains/JetBrainsMono/releases) — download the .zip, extract `.ttf` files
2. Add to the Xcode project target: `JetBrainsMono-Regular.ttf` and `JetBrainsMono-Bold.ttf` (drag into Xcode, "Add to target: CaliTimer" checked)
3. Declare in `Info.plist` under key `UIAppFonts` (Array of Strings): `"JetBrainsMono-Regular.ttf"` and `"JetBrainsMono-Bold.ttf"`
4. Verify PostScript name: after adding, run a quick debug print `UIFont.fontNames(forFamilyName: "JetBrains Mono")` in a temporary view or the app init to confirm the exact PostScript names. Adjust `BrandFonts.swift` if they differ from `JetBrainsMono-Regular` / `JetBrainsMono-Bold`.

- Do NOT use `@Published` or `ObservableObject` — use `@Observable` (iOS 17+)
