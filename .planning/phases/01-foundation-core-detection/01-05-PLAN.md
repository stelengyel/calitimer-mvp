final class Session {

**Personal best update** — insert-or-update pattern:
```swift
private func updatePersonalBest(for skill: String, duration: TimeInterval, date: Date) {
    // Fetch existing PB
    let descriptor = FetchDescriptor<SkillPersonalBest>(
        predicate: #Predicate { $0.skill == skill }
    )
    if let existing = try? modelContext.fetch(descriptor).first {
        if duration > existing.bestDuration {
            existing.bestDuration = duration
            existing.achievedDate = date
        }
    } else {
        let pb = SkillPersonalBest(skill: skill, bestDuration: duration, achievedDate: date)
        modelContext.insert(pb)
    }
}
```
