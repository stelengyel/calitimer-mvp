Implement `VideoFrameReader` exactly per RESEARCH.md Pattern 6, with all performance mandates applied:

```swift
import AVFoundation
import Vision

final class VideoFrameReader {
    struct DetectedHold: Identifiable {
        let id = UUID()
        let startTime: CMTime
        let endTime: CMTime
        let duration: TimeInterval
    }

    // Single Vision request — reused across all frames (performance mandate)
    private let bodyPoseRequest = VNDetectHumanBodyPoseRequest()

    func processVideo(
        at url: URL,
        stride: Int = 3,
        onProgress: ((Double) -> Void)? = nil
    ) async throws -> [DetectedHold]
}
```

**Implementation details — ALL performance mandates must be met:**

1. **AVAssetReader setup:**
   ```swift
   let asset = AVURLAsset(url: url)
   let track = try await asset.loadTracks(withMediaType: .video).first!
   let reader = try AVAssetReader(asset: asset)
   // nil outputSettings = native pixel format (YpCbCr, NOT BGRA)
   // MANDATORY: do NOT pass BGRA settings — iPhone video is natively YpCbCr; BGRA conversion wastes CPU
   let trackOutput = AVAssetReaderTrackOutput(track: track, outputSettings: nil)
   reader.add(trackOutput)
   reader.startReading()
   ```

2. **Frame rate and total frames (for progress):**
   ```swift
   let duration = try await asset.load(.duration)
   let frameRate = try await track.load(.nominalFrameRate)
   let totalFrames = max(1, Int(CMTimeGetSeconds(duration) * Double(frameRate)))
   ```

3. **Stride-adjusted debounce thresholds:**
   At stride=3 (processing every 3rd frame), effective rate = ~10fps. Default HoldStateMachine thresholds (12 detect, 10 loss) are calibrated for 30fps. Divide by stride:
   ```swift
   var stateMachine = HoldStateMachine()
   stateMachine.detectionThreshold = max(1, 12 / stride)  // ~4 at stride 3
   stateMachine.lossThreshold = max(1, 10 / stride)       // ~3 at stride 3
   ```
   MANDATORY per performance mandates: thresholds must be adjusted proportionally for stride.

4. **Frame processing loop:**
   ```swift
   var frameIndex = 0
   var currentHoldStart: CMTime?

   while let sampleBuffer = trackOutput.copyNextSampleBuffer() {
       // Cancellation check every ~30 source frames (~1s at 30fps)
       if frameIndex % 30 == 0 {
           try Task.checkCancellation()
           let progress = min(1.0, Double(frameIndex) / Double(totalFrames))
           onProgress?(progress)
       }
       frameIndex += 1

       // Frame stride — skip frames to reduce Vision calls
       guard frameIndex % stride == 0 else { continue }

       let timestamp = CMSampleBufferGetPresentationTimeStamp(sampleBuffer)

       guard let pixelBuffer = CMSampleBufferGetImageBuffer(sampleBuffer) else { continue }

       // Reuse bodyPoseRequest; create VNImageRequestHandler per frame
       let handler = VNImageRequestHandler(cvPixelBuffer: pixelBuffer, orientation: .up, options: [:])
       try? handler.perform([bodyPoseRequest])

       let classification: HandstandClassifier.Classification
       if let observation = bodyPoseRequest.results?.first {
           classification = classifier.classify(observation)
       } else {
           classification = .insufficientData
       }

       let previousState = stateMachine.state
       stateMachine.feed(classification)

       // Detect timing start
       if case .timing = stateMachine.state, case .detected = previousState {
           currentHoldStart = timestamp
       }
       // Detect hold end
       if case .ended(let duration) = stateMachine.state {
           if let start = currentHoldStart {
               holds.append(DetectedHold(startTime: start, endTime: timestamp, duration: duration))
           }
           currentHoldStart = nil
           stateMachine.reset()
       }
   }
   ```
   Note: `classifier` is a local `HandstandClassifier()` instance (pure struct, no shared state needed).

5. **Final progress report:**
   ```swift
   onProgress?(1.0)
   return holds
   ```

6. **MUST be nonisolated or background-actor-isolated** — mark the method as `nonisolated` so callers on @MainActor use `Task.detached`. The method must never run on MainActor (Pitfall 10).

7. **Temp file note:** The caller (VideoImportView) is responsible for copying the PHPicker URL to a temp directory before calling processVideo. VideoFrameReader accepts any local URL.

**startProcessing(url:):**
```swift
func startProcessing(url: URL) {
    isProcessing = true
    progress = 0
    detectedHolds = []
    error = nil

    processingTask = Task.detached {   // MANDATORY: Task.detached to avoid inheriting MainActor
        do {
            let reader = VideoFrameReader()
            let holds = try await reader.processVideo(
                at: url,
                stride: 3,
                onProgress: { p in
                    Task { @MainActor in progress = p }
                }
            )
            await MainActor.run {
                detectedHolds = holds
                isProcessing = false
            }
        } catch is CancellationError {
            await MainActor.run { isProcessing = false }
        } catch {
            await MainActor.run {
                self.error = error.localizedDescription
                isProcessing = false
            }
        }
        // Clean up temp file after processing
        try? FileManager.default.removeItem(at: url)
    }
}
```

**Cancellation on dismiss:**
```swift
.onDisappear {
    processingTask?.cancel()
}
```
MANDATORY: cancel Task when view disappears to prevent background CPU waste.
